set(SDL2_INCLUDE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/SDL2/include")
set(SDL2_LIBRARY_DIR "${CMAKE_CURRENT_SOURCE_DIR}/SDL2/lib/x64")

set(SDL2_IMAGE_INCLUDE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/SDL2_image/include")
set(SDL2_IMAGE_LIBRARY_DIR "${CMAKE_CURRENT_SOURCE_DIR}/SDL2_image/lib/x64")

set(SDL2_TTF_INCLUDE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/SDL2_ttf/include")
set(SDL2_TTF_LIBRARY_DIR "${CMAKE_CURRENT_SOURCE_DIR}/SDL2_ttf/lib/x64")

set(SDL2_MIXER_INCLUDE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/SDL2_mixer/include")
set(SDL2_MIXER_LIBRARY_DIR "${CMAKE_CURRENT_SOURCE_DIR}/SDL2_mixer/lib/x64")

include_directories(
	${SDL2_INCLUDE_DIR}
	${SDL2_IMAGE_INCLUDE_DIR}
	${SDL2_TTF_INCLUDE_DIR}
	${SDL2_MIXER_INCLUDE_DIR}
)

link_directories(
	${SDL2_LIBRARY_DIR}
	${SDL2_IMAGE_LIBRARY_DIR}
	${SDL2_TTF_LIBRARY_DIR}
	${SDL2_MIXER_LIBRARY_DIR}
)

find_library(SDL2_LIBRARY SDL2 HINTS ${SDL2_LIBRARY_DIR})
find_library(SDL2_MAIN_LIBRARY SDL2main HINTS ${SDL2_LIBRARY_DIR})
find_library(SDL2_IMAGE_LIBRARY SDL2_image HINTS ${SDL2_IMAGE_LIBRARY_DIR})
find_library(SDL2_TTF_LIBRARY SDL2_ttf HINTS ${SDL2_TTF_LIBRARY_DIR})
find_library(SDL2_MIXER_LIBRARY SDL2_mixer HINTS ${SDL2_MIXER_LIBRARY_DIR})

set(SDL2_INCLUDE_DIRS ${SDL2_INCLUDE_DIR} PARENT_SCOPE)
set(SDL2_IMAGE_INCLUDE_DIRS ${SDL2_IMAGE_INCLUDE_DIR} PARENT_SCOPE)
set(SDL2_TTF_INCLUDE_DIRS ${SDL2_TTF_INCLUDE_DIR} PARENT_SCOPE)
set(SDL2_MIXER_INCLUDE_DIRS ${SDL2_MIXER_INCLUDE_DIR} PARENT_SCOPE)

set(SDL2_LIBRARIES ${SDL2_LIBRARY} ${SDL2_MAIN_LIBRARY} PARENT_SCOPE)
set(SDL2_IMAGE_LIBRARIES ${SDL2_IMAGE_LIBRARY} PARENT_SCOPE)
set(SDL2_TTF_LIBRARIES ${SDL2_TTF_LIBRARY} PARENT_SCOPE)
set(SDL2_MIXER_LIBRARIES ${SDL2_MIXER_LIBRARY} PARENT_SCOPE)
